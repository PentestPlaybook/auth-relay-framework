1. Enable SSH Access:

pkg update
pkg install openssh
passwd
sshd
whoami

------------------------------------------------------------------------------------------------------------------------------------
2. Connect via SSH:

cd C:\Users\user\Downloads\platform-tools-latest-windows\platform-tools
.\adb forward tcp:8022 tcp:8022
ssh -p 8022 u0_a377@localhost

------------------------------------------------------------------------------------------------------------------------------------
3. Installations:

pkg install vim
pkg install wget
pkg install libandroid-support
pkg install libandroid-posix-semaphore
pkg install x11-repo
pkg install termux-x11-nightly
pkg install geckodriver
pkg install sshpass

------------------------------------------------------------------------------------------------------------------------------------
4. File Permissions:

su
(Grant Termux Root Permissions in Magisk)
chcon u:object_r:shell_exec:s0 /data/data/com.termux/files/usr/lib/libandroid-support.so
chcon u:object_r:shell_exec:s0 /data/data/com.termux/files/usr/lib/libandroid-posix-semaphore.so
chmod 644 /data/data/com.termux/files/usr/lib/*.so*
chmod 755 /data/data/com.termux/files/usr/bin/*

------------------------------------------------------------------------------------------------------------------------------------

chmod 755 /data/data/com.termux

chmod 755 /data/data/com.termux/files/usr/lib/apt/methods/https
